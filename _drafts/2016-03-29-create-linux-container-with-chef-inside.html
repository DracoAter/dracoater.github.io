We have downloaded an ubuntu image and have it locally.


$ lxc init lxd:ubuntu/trusty/i386 c1
$ lxc start c1
$ lxc exec c1 -- apt-get update
$ lxc exec c1 -- apt-get install -y ruby2.0 ruby2.0-dev build-essential
$ lxc exec c1 -- gem2.0 update --system
$ lxc exec c1 -- gem2.0 install chef:12.4.1 -N --verbose


lxc exec c1 -- apt-get purge build-essential
lxc exec c1 -- apt-get autoremove
lxc exec c1 -- apt-get clean
lxc exec c1 -- apt-get autoclean -y

lxc stop c1
lxc publish c1 --alias=trusty
